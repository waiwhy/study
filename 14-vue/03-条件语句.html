2、else、else-if：demo02

<div id="app">
    <input type="radio" v-model="ok" value="1" />
    <input type="radio" v-model="ok" value="2" />
    <input type="radio" v-model="ok" value="3" />
    <div v-if="ok==1">
        <h1>一一！！</h1>
    </div>
    <div v-if="ok==2">
        <h1>二二！！</h1>
    </div>
    <div v-if="ok==3">
        <h1>三三！！</h1>
    </div>
    <div v-else>
        <h1>四四！！</h1>
    </div>
</div>
<script src="../script/vue.min.js"></script>
<script>
new Vue({
    el: '#app',
    data: {
        ok: 1
    }
});
</script>
