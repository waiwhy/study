测试：ReadExcelHSSF.java

优点：读取少数据，读取速度很快

缺点：读取大数据量，读取速度很慢，会导致内存溢出

内存溢出原因，将流中的数据全部加载到内存，进行输出。

这种读方式是用户驱动模式，由用户指定想获取的数据。

例1：
testRead1：循环读取多个sheet，单元格数据类型是 字符串

package name.yaohuan.poi;
/**
 * 传统方式采用HSSFWorkbook读取xls文件内容，数据量大时报内存溢出
 *
 */
public class ReadExcelHSSF {
    
    @Test
    public void testRead1() throws IOException {
        //文件输入流
        InputStream is = new FileInputStream("d:/poi-excel/商品表-1.xls");
        //创建hssf的workbook，将文件流传入workbook
        HSSFWorkbook workbook = new HSSFWorkbook(is);
        //解析workbook的内容，getNumberOfSheets()得到所有sheet的个数
        for (int numSheet = 0; numSheet < workbook.getNumberOfSheets(); numSheet++) {
            //得到workbook某个 的sheet，numSheet是sheet 的序号，序号从0开始
            HSSFSheet sheet = workbook.getSheetAt(numSheet);
            if (sheet == null) {
                continue;
            }
            //sheet名称
            String sheetName = sheet.getSheetName();
            System.out.println("=====" + sheetName + "=====");
            // 循环行Row
            for (int rowNum = 0; rowNum <= sheet.getLastRowNum(); rowNum++) {
                //读取每一行数据 ,rowNum指定行下标 从0开始
                HSSFRow row = sheet.getRow(rowNum);
                System.out.println("-----row " + row.getRowNum() + "-----");
                //读取单元格内容
                for(int cellNum = 0; cellNum <= row.getLastCellNum(); cellNum++){
                    //读取一行中某个单元格内容，cellNum指定单元格的下标，从0开始
                    HSSFCell cell = row.getCell(cellNum);
                    if(cell == null){
                        continue;
                    }
                    System.out.println(cell.getStringCellValue());
                    //System.out.println(cell.getNumericCellValue());
                }
            }
        }
    }

}

testRead2：读取一个sheet，单元格有多种数据类型
字符串、数值、日期、空值、公式、布尔
    @Test
    public void testRead2() throws IOException {
        //文件输入流
        InputStream is = new FileInputStream("d:/poi-excel/商品表-2-多种数据类型.xls");
        //创建hssf的workbook，将文件流传入workbook
        HSSFWorkbook workbook = new HSSFWorkbook(is);
        HSSFSheet sheet = workbook.getSheetAt(0);
        //公式计算类
        HSSFFormulaEvaluator formulaEvaluator = new HSSFFormulaEvaluator(workbook);
        // 循环行Row
        for (int rowNum = 0; rowNum <= sheet.getLastRowNum(); rowNum++) {
            //读取每一行数据 ,rowNum指定行下标 从0开始
            HSSFRow row = sheet.getRow(rowNum);
            System.out.println("-----row " + row.getRowNum() + "-----");
            //读取单元格内容
            for(int cellNum = 0; cellNum <= row.getLastCellNum(); cellNum++){
                //读取一行中某个单元格内容，cellNum指定单元格的下标，从0开始
                HSSFCell cell = row.getCell(cellNum);
                if(cell == null){
                    continue;
                }
                //下面两行无论使用哪个都报错
                //System.out.println(cell.getNumericCellValue());
                //System.out.println(cell.getStringCellValue());
                //使用下面的方法判断数据类型
                int cellType = cell.getCellType();
                //System.out.println(cellType);
                switch (cellType) {
                case Cell.CELL_TYPE_NUMERIC://0
                    if(HSSFDateUtil.isCellDateFormatted(cell)){
                        SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
                        System.out.println(sdf.format(HSSFDateUtil.getJavaDate(cell.getNumericCellValue())));
                        System.out.println(sdf.format(cell.getDateCellValue()));
                    }else{
                        //DecimalFormat df = new DecimalFormat("#");
                        //System.out.println(df.format(cell.getNumericCellValue()));
                        System.out.println(cell.getNumericCellValue());
                    }
                    break;
                case Cell.CELL_TYPE_STRING://1
                    System.out.println(cell.getStringCellValue());
                    break;
                case Cell.CELL_TYPE_FORMULA://2
                    //得到公式
                    System.out.println(cell.getCellFormula());
                    CellValue evaluate = formulaEvaluator.evaluate(cell);
                    int evaluateCellType = evaluate.getCellType();
                    switch (evaluateCellType) {
                    case Cell.CELL_TYPE_NUMERIC:
                        System.out.println(cell.getNumericCellValue());
                        break;
                    case Cell.CELL_TYPE_BOOLEAN:
                        System.out.println(cell.getBooleanCellValue());
                        break;
                    }
                    break;
                case Cell.CELL_TYPE_BLANK://3
                    //注意空和没有值不一样，从来没有录入过内容的单元格不属于任何数据类型，不会走这个case
                    System.out.println("空");
                    //System.out.println(cell.getStringCellValue());
                    break;
                case Cell.CELL_TYPE_BOOLEAN://4
                    System.out.println(cell.getBooleanCellValue());
                    break;
                case Cell.CELL_TYPE_ERROR:
                    System.out.println("数据类型错误");
                    break;
                default:
                    System.out.println(cellType + "：未知数据类型");
                    break;
                }
            }
        }
    }
