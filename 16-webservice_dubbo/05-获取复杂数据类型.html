2、创建service
package com.qfedu.ws_server_gp;
import java.util.ArrayList;
import java.util.List;
import javax.jws.WebService;
import com.qfedu.ws_server_gp.model.Person;
@WebService(targetNamespace="ws.yaohuan.com")
public class PersonService {
    List<Person> personList = new ArrayList<Person>();
    
    public void addPerson(Person person) {
        personList.add(person);
    }
    
    public List<Person> getPersonAll(){
        return personList;
    }
}



3、创建服务发布类
package com.qfedu.ws_server_gp;
import javax.xml.ws.Endpoint;
public class PersonServicePublish {
    public static void main(String[] args) {
        
        //jdk发布一个webservice服务
        Endpoint.publish("http://127.0.0.1:8088/person", new PersonService());
    }
}

6、定义客户端
package com.qfedu.client;
import java.util.List;
import com.yaohuan.ws.Person;
import com.yaohuan.ws.PersonService;
import com.yaohuan.ws.PersonServiceService;
public class PersonClient {
    public static void main(String[] args) {
        
        PersonServiceService pss = new PersonServiceService();
        PersonService ps = pss.getPersonServicePort();
        
        Person p1 = new Person();
        p1.setId(1);
        p1.setAge(18);
        p1.setName("Tom");
        
        Person p2 = new Person();
        p2.setId(2);
        p2.setAge(20);
        p2.setName("Peter");
        
        ps.addPerson(p1);
        ps.addPerson(p2);
        
        List<Person> personAll = ps.getPersonAll();
        for (Person person : personAll) {
            System.out.println(person.getId() + " " + person.getName() + " " + person.getAge());
        }
    }
}
