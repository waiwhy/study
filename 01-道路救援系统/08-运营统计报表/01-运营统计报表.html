1、SQL分析
SELECT
r.dispatch_vid AS dispatchVid,
v.number,
r.dispatch_plate_number AS dispatchPlateNumber,
r.dispatch_name AS dispatchName, 
SUM(r.back_cost) AS cost,
SUM(r.back_mileage) AS mileage,
SUM(r.back_mileage) * 1.3 AS oilwear

FROM ar_rescue r
LEFT JOIN ar_vehicle v
ON r.dispatch_vid = v.vid

-- where 日期范围
WHERE dispatch_time >= '2016-11-01 00:00:00'
AND dispatch_time <= '2016-11-31 23:59:59'
-- group by
GROUP BY r.dispatch_vid,
r.dispatch_did -- 个别车辆中途更换了驾驶员   group by原理,字段数据相同分为一组

ORDER BY v.number ASC

2、dao层实现
public class ReportDao {
    DataSource dataSource = DataSourceUtil.getDataSource();
    public List<ReportResult> getRescueReportList(String beginDate, String endDate, Float oilwear) throws SQLException {
        Connection connection = DataSourceUtil.getConnection();
        QueryRunner runner = new QueryRunner(dataSource);
        String sql = "SELECT"
                +" r.dispatch_vid AS dispatchVid,"
                +" v.number,"
                +" r.dispatch_plate_number AS dispatchPlateNumber,"
                +" r.dispatch_name AS dispatchName, "
                +" SUM(r.back_cost) AS cost,"
                +" SUM(r.back_mileage) AS mileage,"
                +" SUM(r.back_mileage) * ? AS oilwear"
                +" FROM ar_rescue r"
                +" LEFT JOIN ar_vehicle v"
                +" ON r.dispatch_vid = v.vid"
                // -- where 日期范围
                +" WHERE dispatch_time >= ?"
                +" AND dispatch_time <= ?"
                // -- group by
                +" GROUP BY r.dispatch_vid,"
                // -- 个别车辆中途更换了驾驶员
                +" r.dispatch_did "
                +" ORDER BY v.number ASC";
        
        //System.out.println(sql);
        return runner.query(
                connection, sql, new BeanListHandler<ReportResult>(ReportResult.class),
                oilwear, beginDate, endDate);
    }
}

5、controller
public class ReportController extends BaseServlet {
    private static final long serialVersionUID = 1L;
   
    private ReportService target = new ReportService();
    private ReportService reportService = (ReportService)new ProxyFactory(target).getProxyInstance();
    public void list (HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        RequestDispatcher view = request.getRequestDispatcher("/WEB-INF/jsp/report/list.jsp");
        view.forward(request, response);
    }
    
    public void list_data (HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException, SQLException {
            
        String beginDate = request.getParameter("beginTime") + "00:00:00";
        String endDate = request.getParameter("endTime") + "23:59:59";
        String oilwearStr = request.getParameter("oilwear");
        Float oilwear = Float.parseFloat(oilwearStr);
        List<ReportResult> rescueReportList = reportService.getRescueReportList(beginDate, endDate, oilwear);
        String json = JSON.toJSONString(rescueReportList);
        response.getWriter().print(json);//http请求的响应数据
    }
}

7、页面初始化以及查询表单的实现
注意：表单的提交调用了bootstraptable的表格初始化代码，表格数据的获取方式配制成了post
默认情况下，bootstraptable的post的content-type的值是“application/json ”，我们需要将它改成“ x-www-form-urlencoded”
    <script>
    
    $(function(){
        
        initSearchForm();
        
        $('#genReport').click(function(){
            if($('#myform').valid()){
                loadTable();
            }
        });
    });    
    
    function initSearchForm(){
        
         var option = {
            lang: 'ch',
            format: 'Y-m-d',
            timepicker: false
        };
        $('#beginTime').datetimepicker(option);
        $('#endTime').datetimepicker(option);
        
        $('#myform').validate({
            rules: {
                beginTime: {
                    required: true
                },
                endTime: {
                    required: true
                },
                oilwear: {
                    required: true,
                    number: true,
                    min: 0
                   }
            },
            messages:{
                beginTime: {
                    required: '请输入开始时间'
                },
                endTime: {
                    required: '请输入结束时间'
                },
                oilwear: {
                    required: '请输入油耗',
                    number: '请输入合法的数字',
                    min: '请输入合法的数字'
                   }
            }
        });
    }
    
    function loadTable(){
        $('#table').bootstrapTable('destroy');
        $('#table').bootstrapTable({
            url: 'report?method=list_data',
            method: 'post',
            contentType: 'application/x-www-form-urlencoded; charset=UTF-8',
            queryParams: {
                oilwear: $('#oilwear').val(),
                beginTime: $('#beginTime').val(),
                endTime: $('#endTime').val()
            },
            showRefresh: true,
            toolbar: '#toolbar',
            showFooter: true,
            columns: [{
                field: 'number',
                title: '车辆编号'
            }, {
                field: 'dispatchPlateNumber',
                title: '车牌号码'
            }, {
                field: 'dispatchName',
                title: '驾驶员',
                footerFormatter: function (value) {
                    return '<span class="pull-right">总计：</span>';
                }
            }, {
                field: 'cost',
                align: 'right',
                title: '救援收入(元)', 
                formatter: function(value){
                    return value.toFixed(2);
                },
                footerFormatter: function (rowList) {
                    var sum = 0;
                    $.each(rowList, function(i, row){
                        console.log(row.cost);
                        sum += row.cost;
                    });
                    return sum.toFixed(2);
                }
            }, {
                field: 'mileage',
                align: 'right',
                title: '行驶里程(公里)', 
                formatter: function(value){
                    return value.toFixed(2);
                },
                footerFormatter: function (rowList) {
                    var sum = 0;
                    $.each(rowList, function(i, row){
                        sum += row.mileage;
                    });
                    return sum.toFixed(2);
                }
            }, {
                field: 'oilwear',
                align: 'right',
                title: '油耗费用(元)', 
                formatter: function(value){
                    return value.toFixed(2);
                },

                        sum += row.oilwear;
                    });
                    return sum.toFixed(2);
                }
            }]
        });
