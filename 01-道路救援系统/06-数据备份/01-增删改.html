package com.qfedu.arrescue.dao;
public class BackupDao {
    DataSource dataSource = DataSourceUtil.getDataSource();
    
    public List<Backup> getBackupList() throws SQLException{
        
        Connection connection = DataSourceUtil.getConnection();
        QueryRunner runner = new QueryRunner(dataSource);
        String sql = "SELECT "
                + " bid, "
                + " createtime, "
                + " filepath "
                + " FROM ar_backup order by bid desc";
        return runner.query(connection, sql, new BeanListHandler<Backup>(Backup.class));
    }
    
    public Backup getBackupById(String id) throws SQLException{
        
        Connection connection = DataSourceUtil.getConnection();
        QueryRunner runner = new QueryRunner(dataSource);
        String sql = "SELECT "
                + " bid, "
                + " createtime, "
                + " filepath "
                + " FROM ar_backup "
                + " where bid = ?";
        return runner.query(connection, sql, new BeanHandler<Backup>(Backup.class), id);
    }
    
    public int deleteMoreByIds(String[] ids) throws SQLException {
        Connection connection = DataSourceUtil.getConnection();
        QueryRunner runner = new QueryRunner(dataSource);
        String idStr = StringUtil.genCommaSeperationStr(ids);
        String sql = "delete from ar_backup where bid in ( " + idStr + " )";
        return runner.update(connection, sql);
    }
    public int deleteById(String id) throws SQLException {
        Connection connection = DataSourceUtil.getConnection();
        QueryRunner runner = new QueryRunner(dataSource);
        String sql = "delete from ar_backup where bid = ?";
        return runner.update(connection, sql, id);
    }
    public int save(Backup backup) throws SQLException {
        Connection connection = DataSourceUtil.getConnection();
        QueryRunner runner = new QueryRunner(dataSource);
        String sql = " INSERT INTO ar_backup ( "
                + " filepath,"
                + " createtime"
                + " ) VALUES (?,?)" ;
        return runner.update(connection, sql, 
                backup.getFilepath(),
                backup.getCreatetime());
    }
}

package com.qfedu.arrescue.controller;
public class BackupServlet extends BaseServlet {
    private static final long serialVersionUID = 1L;
    //目标对象
    BackupService target = new BackupService();
    //代理对象
    BackupService backupService = (BackupService)new ProxyFactory(target).getProxyInstance();
    public void index(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        RequestDispatcher view = request.getRequestDispatcher("/WEB-INF/jsp/backup/list.jsp");
        view.forward(request, response);
    }
    
    public void list(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException, SQLException {
        List<Backup> backupList = backupService.getBackupList();
        String json = JSON.toJSONString(backupList);
        response.getWriter().print(json);
    }
    
    public void deleteById(HttpServletRequest request, HttpServletResponse response)throws ServletException, IOException, SQLException {
        String id = request.getParameter("id");
        ResultBean reasult = backupService.deleteById(id);
        
        String json = JSON.toJSONString(reasult);
        response.getWriter().print(json);
    }
    
    public void deleteMore(HttpServletRequest request, HttpServletResponse response)throws ServletException, IOException, SQLException{
        String[] ids = request.getParameterValues("ids");
        ResultBean reasult = backupService.deleteMoreByIds(ids);
        String json = JSON.toJSONString(reasult);
        response.getWriter().print(json);
    }
    
    public void backup(HttpServletRequest request, HttpServletResponse response) throws Exception {
        
        ResultBean reasult = backupService.save();
        
        String json = JSON.toJSONString(reasult);
        response.getWriter().print(json);
    }
    
    public void restore(HttpServletRequest request, HttpServletResponse response) throws Exception {
        
        String id = request.getParameter("id");
        ResultBean reasult = backupService.restore(id);
        
        String json = JSON.toJSONString(reasult);
        response.getWriter().print(json);
    }
}

    function deleteItem(id){
        
        layer.confirm('您确定要删除所选记录吗？', {

                    layer.alert('删除成功');

    function deleteMoreItem() {  
        
         //返回所有选择的行，当没有选择的记录时，返回一个空数组  
        var rows = $('#table').bootstrapTable('getSelections');   
        if (rows.length == 0) {  
            layer.alert('请选择要删除的数据');  
            return;  
        }  
        //提示确认框
        layer.confirm('您确定要删除所选数据吗？', {
              btn: ['确定', '取消'] //可以无限个按钮
        }, function(index, layero){
            var ids = new Array();  
            //遍历所有选择的行数据，取每条数据对应的ID  
            $.each(rows, function(i, row) { 
                ids[i] = 'ids=' + row['bid'];
            });  
             
            $.ajax({ 
                url : 'backup?method=deleteMore&' + ids.join('&'),
                success : function(data) {  
                    data = JSON.parse(data);
                    if(data.status == 'SUCCESS'){
                        layer.alert('删除成功'); 
                        $('#table').bootstrapTable('refresh');
                    }else{
                        layer.alert('删除失败'); 
                    }
                },  
                error : function() {  
                    layer.alert('服务器没有返回数据，可能服务器忙，请重试');
