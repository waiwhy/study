二、编辑
1、controller
    public void edit(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException, SQLException {
        
        //获取被编辑的id
        String id = request.getParameter("id");
        
        //获取当前用户
        User user = userService.getUserById(id);
        
        //获取角色列表
        List<Role> roleList = roleService.getRoleList();
        
        request.setAttribute("user", user);
        request.setAttribute("roleList", roleList);
        RequestDispatcher view = request.getRequestDispatcher("/WEB-INF/jsp/user/edit.jsp");
        view.forward(request, response);
    }

2、jsp
        <div class="form-group">
            <label class="col-sm-4 control-label">角色</label>
            <div class="col-sm-4">
                <select class="form-control" name="rid" id="rid">
                    <c:forEach items="${roleList}" var="role">
                        <c:if test="${user.rid==role.rid}"><option value="${role.rid}" selected>${role.alias}</option></c:if>
                        <c:if test="${user.rid!=role.rid}"><option value="${role.rid}">${role.alias}</option></c:if>
                    </c:forEach>
                </select>
            </div>
        </div>
