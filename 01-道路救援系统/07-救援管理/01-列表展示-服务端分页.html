5、创建service
（1）封装返回值对象
public class BootstrapTableResult {
    private long total;
    private List<?> rows;
}
（2）service
public class RescueService {
    private RescueDao rescueDao = new RescueDao();
    
    public ResultBean getRescueList(int offset, int pageSize) throws SQLException{
        
        List<Rescue> rescueList = rescueDao.getRescueList(offset, pageSize);
        long total = rescueDao.getTotal();
        
        BootstrapTableResult bootstrapTableResult = new BootstrapTableResult();
        bootstrapTableResult.setRows(rescueList);
        bootstrapTableResult.setTotal(total);
        
        return ResultBean.success(bootstrapTableResult);
    }
}

6、创建controller
public class RescueServlet extends BaseServlet {
    private static final long serialVersionUID = 1L;
    private RescueService target = new RescueService();
    private RescueService rescueService = (RescueService)new ProxyFactory(target).getProxyInstance();
    public void index (HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        RequestDispatcher view = request.getRequestDispatcher("/WEB-INF/jsp/rescue/list.jsp");
        view.forward(request, response);
    }

    public void list(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException, SQLException {
        String offsetStr = request.getParameter("offset");
        String limitStr = request.getParameter("limit");
        
        int pageSize = 10;
        int offset = 0;
        if(StringUtils.isNotEmpty(offsetStr)) {
            offset = Integer.parseInt(offsetStr);
        }
        if(StringUtils.isNotEmpty(limitStr)) {
            pageSize = Integer.parseInt(limitStr);
        }
        
        ResultBean resultBean = rescueService.getRescueList(offset, pageSize);
        String json = JSON.toJSONString(resultBean.getData());
        response.getWriter().print(json);
    }
}
