1、使用ajax进行表单提交
注意需要表单的同步加载，设置 async:false
    function saveStep2(){
        if(!$('#form2').valid()){
            return false;
        }
        
        var isok = false;
        $('#form2').ajaxSubmit({
            async: false,//******同步加载数据*******
            url: 'rescue?method=saveStep2&rid=' + $('#rid').val(),
            success: function(result){
                result = JSON.parse(result);
                if(result.status){
                    layer.alert('保存成功');
                    isok = true;
                }else{
                    layer.alert('保存失败');
                }
            },
            error: function(){
                layer.alert('远程服务器连接失败，请稍候再试！');
            }
        });
        
        return isok;
    }

2、controller
    public void saveStep2(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException, IllegalAccessException, InvocationTargetException, SQLException {
        String rid = request.getParameter("rid");
        
        Map<String, String[]> parameterMap = request.getParameterMap();
        Rescue rescue = new Rescue();
        
        //自己指定一个类型转换器（将String转成Date）
        ConvertUtils.register(new DateConverter("yyyy-MM-dd HH:mm"), Date.class);
        //映射封装
        BeanUtils.populate(rescue, parameterMap);
        rescue.setRid(Integer.parseInt(rid));
        ResultBean resultBean = rescueService.updateStepBackByRid(rescue);
        String json = JSON.toJSONString(resultBean);
        response.getWriter().print(json);//http请求的响应数据
    }

3、service
    public ResultBean updateStepBackByRid(Rescue rescue) throws SQLException {
        //保存
        rescue.setRstatus(3);
        int count = rescueDao.updateStepBackByRid(rescue);
        return count > 0 ? ResultBean.success(rescue) : ResultBean.error();
    }
