    public void saveStep0(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException, IllegalAccessException, InvocationTargetException, SQLException {
        Map<String, String[]> parameterMap = request.getParameterMap();
        Rescue rescue = new Rescue();
        BeanUtils.populate(rescue, parameterMap);
        //获取调度员信息
        HttpSession session = request.getSession();
        User user = (User)session.getAttribute("user");
        if(user != null) {
            rescue.setDispatchUid(user.getUid());
            rescue.setDispatchUsername(user.getUsername());
        }
        ResultBean resultBean = rescueService.save(rescue);
        String json = JSON.toJSONString(resultBean);
        response.getWriter().print(json);//http请求的响应数据
    }

10、service
生成流水号并存储
获取最后插入的id
    public ResultBean save(Rescue rescue) throws SQLException {
        
        //生成订单编号
        SerialNumberUtil numberUtil = new SerialNumberUtil();
        String param = numberUtil.getPrefix() + new DateTime().toString(numberUtil.getTimeFormat()) + "%";//'HL20170215%';
        String currentSerialNumber = rescueDao.getLastSerialNumber(param);
        String serialNum = numberUtil.getSerialNum(currentSerialNumber);
        rescue.setDispatchNumber(serialNum);
        //保存
        rescue.setRstatus(1);
        rescue.setApplyTime(new Date());
        int count = rescueDao.save(rescue);
        //获取最后一个id
        Integer lastId = rescueDao.getLastId();
        rescue.setRid(lastId);
        
        return count > 0 ? ResultBean.success(rescue) : ResultBean.error();
    }

11、dao层
    public String getLastSerialNumber(String param) throws SQLException {
        Connection connection = DataSourceUtil.getConnection();
        QueryRunner runner = new QueryRunner(dataSource);
        String sql = "SELECT dispatch_number "
                + " FROM ar_rescue "
                + " WHERE dispatch_number "
                + " LIKE ? "
                + " ORDER BY dispatch_number DESC "
                + " LIMIT 1";
        String num = runner.query(connection, sql, new ScalarHandler<String>(1), param);
        return (String)num;
    }
    public int save(Rescue rescue) throws SQLException {
        Connection connection = DataSourceUtil.getConnection();
        QueryRunner runner = new QueryRunner(dataSource);
        String sql = " INSERT INTO ar_rescue ( "
                + " rstatus,"
                + " apply_time,"
                + " apply_customer,"
                + " apply_owner,"
                + " apply_tel,"
                + " apply_reason,"
                + " apply_place,"
                + " apply_plate_number,"
                + " apply_type,"
                + " apply_color,"
                + " apply_destination,"
                + " apply_destination_type,"
                + " apply_cost,"
                + " apply_common,"
                + " dispatch_uid,"
                + " dispatch_username,"
                + " dispatch_number"
                + " ) VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)" ;
        return runner.update(connection, sql, 
                rescue.getRstatus(),
                rescue.getApplyTime(),
                rescue.getApplyCustomer(),
                rescue.getApplyOwner(),
                rescue.getApplyTel(),

    }
    
    public Integer getLastId() throws SQLException {
        Connection connection = DataSourceUtil.getConnection();
        QueryRunner runner = new QueryRunner(dataSource);
        String sql = "SELECT LAST_INSERT_ID()";
        BigInteger id = runner.query(connection, sql, new ScalarHandler<BigInteger>(1));
        return id.intValue();
    }
