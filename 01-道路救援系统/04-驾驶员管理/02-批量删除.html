    $(function(){
        $('#delete_more').click(function(){
            deleteMoreItem();
        });
    });    

    function deleteMoreItem() {  
        
         //返回所有选择的行，当没有选择的记录时，返回一个空数组  
        var rows = $('#table').bootstrapTable('getSelections');   
        if (rows.length == 0) {  
            layer.alert('请选择要删除的数据');  
            return;  
        }  
        //提示确认框
        layer.confirm('您确定要删除所选数据吗？', {
              btn: ['确定', '取消'] //可以无限个按钮
        }, function(index, layero){
            var url = 'driver?method=deleteMore';
            var ids = new Array();  
            //遍历所有选择的行数据，取每条数据对应的ID  
            $.each(rows, function(i, row) { 
                ids[i] = 'dids=' + row['did'];
            });  
             
            $.ajax({ 
                url : 'driver?method=deleteMore&' + ids.join('&'),
                success : function(data) {  
                    data = JSON.parse(data);
                    if(data.status == 'SUCCESS'){
                        layer.alert('删除成功'); 
                        $('#table').bootstrapTable('refresh');
                    }else{
                        layer.alert('删除失败'); 
                    }
                },  
                error : function() {  
                    layer.alert('服务器没有返回数据，可能服务器忙，请重试');  
                }  
            }); 
        });
        
    }

3、创建ResultBean
package com.hljiuyuan.automotiverescue.vo;
public class ReasultBean {
    public static final String SUCCESS = "SUCCESS";
    public static final String ERROR = "ERROR";
    private String status;
    
    public ReasultBean() {
        super();
    }
    public String getStatus() {
        return status;
    }
    public void setStatus(String status) {
        this.status = status;
    }
    
    public static ReasultBean success() {
        ReasultBean result = new ReasultBean();
        result.setStatus(SUCCESS);
        return result;
    }
    
    public static ReasultBean error() {
        ReasultBean result = new ReasultBean();
        result.setStatus(ERROR);
        return result;
    }
}
