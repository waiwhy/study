7、controller（显示和数据）
public class DriverServlet extends BaseServlet {
    private static final long serialVersionUID = 1L;
      
    //目标对象
    DriverService target = new DriverService();
    //代理对象
    DriverService driverService = (DriverService)new ProxyFactory(target).getProxyInstance();
    
    public void index(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        RequestDispatcher view = request.getRequestDispatcher("/WEB-INF/jsp/driver/list.jsp");
        view.forward(request, response);
    }
    
    public void list(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException, SQLException {
        List<Driver> driverList = driverService.getDriverList();
        String json = JSON.toJSONString(driverList);
        response.getWriter().print(json);
    }
}
