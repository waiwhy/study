2、dao层做关联查询
分析：使用外连接
-- 内连接
SELECT * FROM ar_user u, ar_role r WHERE u.rid = r.rid
-- 外连接
SELECT * FROM ar_user u LEFT JOIN ar_role r ON u.rid = r.rid;
    public List<UserCustom> getUserList(int offset, int pageSize) throws SQLException {
        
        Connection connection = DataSourceUtil.getConnection();
        QueryRunner runner = new QueryRunner(dataSource);
        String sql = "select"
                + " u.uid,"
                + " u.username,"
                + " u.upassword,"
                + " u.realname,"
                + " u.email,"
                + " u.telephone,"
                + " u.last_login_time as lastLoginTime,"//处理属性名和列名不一致
                + " u.locked, "
                + " r.rname as rolename "
                + " from ar_user u "
                + " left join ar_role r "
                + " on u.rid = r.rid"
                + " limit ?, ? ";
        return runner.query(connection, sql, new BeanListHandler<UserCustom>(UserCustom.class), offset, pageSize);
    }

补充：多表连接
-- 内连接
SELECT * FROM ar_user u, ar_role r, ar_type t WHERE u.rid = r.rid AND u.tid = t.tid;

-- 外连接
SELECT * 
FROM ar_user u 
LEFT JOIN ar_role r ON u.rid = r.rid 
LEFT JOIN ar_type t ON u.tid = t.tid;
