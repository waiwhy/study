    public PageBean getUserList(int currentPage, int pageSize) throws MyException, SQLException{
        //用户列表
        int offset = (currentPage - 1) * pageSize;
        List<User> userList = userDao.getUserList(offset, pageSize);
        
        Iterator<User> iterator = userList.iterator();
        while(iterator.hasNext()) {
            User user = iterator.next();
            Integer rid = user.getRid();
            Role role = roleDao.getRoleById(rid.toString());
            user.setRole(role);
        }
        
        //总记录数
        long totalNum = userDao.getTotal();
        
        //计算总页数
        long totalPages = totalNum / pageSize;
        if(totalNum % pageSize != 0) {
            totalPages++;
        }
        
        PageBean pageBean = new PageBean();
        pageBean.setDataList(userList);
        pageBean.setCurrentPage(currentPage);
        pageBean.setPageSize(pageSize);
        pageBean.setTotalCount(totalNum);
        pageBean.setTotalPage(totalPages);
        
        return pageBean;
    }
