 2、配置文件：application.properties
该配置文件主要是配置了对mysql的支持，包括用户名、密码、驱动、url等、还有基于Hibernate对jpa的支持特性。
spring.datasource.driver-class-name=com.mysql.jdbc.Driver
spring.datasource.username=root
spring.datasource.password=123456
spring.datasource.url=jdbc:mysql://localhost:3306/spring-jpa?useUnicode=true&characterEncoding=UTF-8
#配置Hibernate的基本属性
spring.jpa.show-sql=true
spring.jpa.hibernate.ddl-auto=update

4、实体类

jpa对创建的实体类有以下基本的要求

  1. 提供一个默认的构造方法
  2. 必须声明主键的ID

package name.yaohuan.springjpa.pojo;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
@Entity
public class Employee {
    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Integer id;
    @Column
    private String name;
    @Column
    private String photo;
    @Column
    private String address;
    public Employee(String name, String photo, String address) {
        this.name = name;
        this.photo = photo;
        this.address = address;
    }
    public String getAddress() {
        return address;
    }
    public void setAddress(String address) {
        this.address = address;
    }
    public Employee() {
    }
    public Integer getId() {
        return id;
    }
    public void setId(Integer id) {
        this.id = id;
    }
    public String getName() {
        return name;
    }
    public void setName(String name) {
        this.name = name;
    }
    public String getPhoto() {
        return photo;
    }
    public void setPhoto(String photo) {
        this.photo = photo;
    }
    @Override
    public String toString() {
        return "Employee{" +
                "id=" + id +
                ", name='" + name + '\'' +
                ", photo='" + photo + '\'' +
                ", address='" + address + '\'' +
                '}';
    }
}

5、声明EmployeeRepository 继承 CrudRepository接口（dao层）
声明该接口不需要实现任何的方法，因为CrudRepository接口定义了常用的方法，包括了添加、删除、修改、查询等操作。
package name.yaohuan.springjpa.repository;
import org.springframework.data.repository.CrudRepository;
import name.yaohuan.springjpa.pojo.Employee;
public interface EmployeeRepository extends CrudRepository<Employee, Integer>{

}

package name.yaohuan.springjpa.service;
import name.yaohuan.springjpa.pojo.Employee;
public interface EmployeeService {
    Employee save(Employee employee);
    Iterable<Employee> saveMulti(Iterable<Employee> iterable);
    Employee findOne(Integer id);
    boolean exists(Integer id);
    Iterable<Employee> findAll();
    Iterable<Employee> findAll(Iterable<Integer> ids);
    long count();
    void delete(Integer id);
    void delete(Employee employee);
    void delete(Iterable<Employee> employees);

}

7、定义service实现类
package name.yaohuan.springjpa.service.impl;
import java.io.Serializable;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import name.yaohuan.springjpa.pojo.Employee;
import name.yaohuan.springjpa.repository.EmployeeRepository;
import name.yaohuan.springjpa.service.EmployeeService;
@Service
public class EmployeeServiceImp<T, ID extends Serializable> implements EmployeeService {
    @Autowired
    private EmployeeRepository repository;
    @Override
    public Employee save(Employee employee) {
        return repository.save(employee);
    }
    @Override
    public Iterable<Employee> saveMulti(Iterable<Employee> iterable) {
        return repository.save(iterable);
    }
    @Override
    public Employee findOne(Integer id) {
        return repository.findOne(id);
    }
    @Override
    public boolean exists(Integer id) {
        return repository.exists(id);
    }
    
    @Override
    public Iterable<Employee> findAll() {
        return repository.findAll();
    }
    @Override
    public Iterable<Employee> findAll(Iterable<Integer> ids) {
        return repository.findAll(ids);
    }
    @Override
    public long count() {
        return repository.count();
    }
    @Override
    public void delete(Integer id) {
        repository.delete(id);
    }
    @Override
    public void delete(Employee employee) {
        repository.delete(employee);
    }
    @Override
    public void delete(Iterable<Employee> employees) {
        repository.delete(employees);
    }
    @Override
    public void deleteAll() {
        repository.deleteAll();
    }
}

package name.yaohuan.springjpa;
import java.util.ArrayList;

import org.junit.Test;
import org.junit.runner.RunWith;

import org.slf4j.LoggerFactory;

import org.springframework.test.context.junit4.SpringRunner;
import name.yaohuan.springjpa.pojo.Employee;
import name.yaohuan.springjpa.service.EmployeeService;
@RunWith(SpringRunner.class)
@SpringBootTest
public class EmployeeServiceTest {
    private static final Logger LOGGER = LoggerFactory.getLogger(EmployeeServiceTest.class);
       
       @Autowired
       private EmployeeService service;
       
       @Test
       public void save() {
          Employee employee = new Employee("Helen", "1.jpg", "Beijing");
          service.save(employee);
       }
       
       @Test
       public void saveMulti(){

          list.add(new Employee("jack","myphoto","北京"));

            service.saveMulti(list);
       }
      
       @Test
       public void findone(){

          LOGGER.info(employee.toString());
       }

       }
      
       @Test

          iterable.forEach(new Consumer<Employee>() {

       }
