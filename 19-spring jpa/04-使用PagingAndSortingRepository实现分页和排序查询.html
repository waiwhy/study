1、定义实体类
package name.yaohuan.springjpa.pojo;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
@Entity
public class Product {
    
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Integer id;
    @Column
    private Float price;
    @Column
    private String title;
    @Column
    private String picUrl;
    @Column
    private Float sellingPrice;
    @Column
    private Integer salesVolume;//销量
    
    public Product() {
    }
    public Product(Float price, String title, String picUrl, Float sellingPrice, Integer salesVolume) {
        super();
        this.price = price;
        this.title = title;
        this.picUrl = picUrl;
        this.sellingPrice = sellingPrice;
        this.salesVolume = salesVolume;
    }
    public Integer getId() {
        return id;
    }
    public void setId(Integer id) {
        this.id = id;
    }
    public Float getPrice() {
        return price;
    }
    public void setPrice(Float price) {
        this.price = price;
    }
    public String getTitle() {
        return title;
    }
    public void setTitle(String title) {
        this.title = title;
    }
    public String getPicUrl() {
        return picUrl;
    }
    public void setPicUrl(String picUrl) {
        this.picUrl = picUrl;
    }
    public Float getSellingPrice() {
        return sellingPrice;
    }
    public void setSellingPrice(Float sellingPrice) {
        this.sellingPrice = sellingPrice;
    }
    public Integer getSalesVolume() {
        return salesVolume;
    }
    public void setSalesVolume(Integer salesVolume) {
        this.salesVolume = salesVolume;
    }
    @Override
    public String toString() {
        return "Product [id=" + id + ", price=" + price + ", title=" + title + ", picUrl=" + picUrl + ", sellingPrice="
                + sellingPrice + ", salesVolume=" + salesVolume + "]";
    }
}

import org.springframework.data.domain.Page;

    Iterable<Product> findProductBySort(Sort sort);
}

4、定义 ProductServiceImpl 实现类
package name.yaohuan.springjpa.service.impl;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Service;
import name.yaohuan.springjpa.pojo.Product;
import name.yaohuan.springjpa.repository.ProductRepository;
import name.yaohuan.springjpa.service.ProductService;
@Service
public class ProductServiceImp  implements ProductService {
    @Autowired
    private ProductRepository repository;
    @Override
    public Product save(Product product) {
        return repository.save(product);
    }
    
    @Override
    public Page<Product> findProductByPage(Pageable pageable) {
        return repository.findAll(pageable);
    }

    @Override
    public Iterable<Product> findProductBySort(Sort sort) {
        return repository.findAll(sort);
    }
}

package name.yaohuan.springjpa;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.PageRequest;
import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
import name.yaohuan.springjpa.pojo.Product;
import name.yaohuan.springjpa.service.ProductService;
@RunWith(SpringJUnit4ClassRunner.class)
@SpringBootTest
public class ProductServiceTest {
    private static Logger logger = LoggerFactory.getLogger(ProductServiceTest.class);
    @Autowired
    private ProductService service;
    @Test
    public void save() {
        Product product = new Product(100F, "手机", "www.baidu.com", 200F, 10);
        service.save(product);
    }
    @Test
    public void showPage() {
        //参数分别表示当前页、每页显示的记录条数
        PageRequest request = new PageRequest(0, 7);
        Page<Product> pages = service.findProductByPage(request);
        int currentPage = pages.getNumber() + 1;
        //判断是否有下一页
        logger.info("--->>"+pages.hasNext());
        //判断是否有上一页
        logger.info("--->>"+pages.hasPrevious());
        //判断是否是第一页
        logger.info("--->>"+pages.isFirst());
        //判断是否是最后一页
        logger.info("--->>"+pages.isLast());
        logger.info("-记录总条数->>" + pages.getTotalElements());
        logger.info("-总页数-->>" + pages.getTotalPages());
        logger.info("--当前页->>" + currentPage);
        logger.info("--->>" + pages.getNumberOfElements());
        logger.info("--->>" + pages.getSize());
        logger.info("--打印输出结果-->>"+pages.getContent());
    }

    @Test
    public  void sort(){
        //按照价格排序
        Sort sort = new Sort(Sort.Direction.ASC,"price");
        Iterable<Product> iterable = service.findProductBySort(sort);
        iterable.forEach(new Consumer<Product>() {
            @Override
            public void accept(Product product) {
                logger.info("-->>"+product);
            }
        });
    }
}
