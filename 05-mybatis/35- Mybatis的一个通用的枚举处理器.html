Mybatis的一个通用的枚举处理器


mybatis本身提供了EnumTypeHandler和EnumOrdinalTypeHandler用来做枚举类型处理，但是不是很灵活，EnumTypeHandler是把枚举的名字和枚举之间做了一个映射，EnumOrdinalTypeHandler是把枚举的ordinary也就是顺序和枚举之间做了一个映射。超出这两种情况，就需要自己动手了。

本文的实现效果：

查询1：

[java] view plain copy
  1. List<User> selectByUserType(UserType userType);  

[html] view plain copy
  1. <select id="selectByUserType" resultMap="BaseResultMap">  
  2.     select * from user_enum where user_type=#{userType,typeHandler=com.chrhc.mybatis.handler.EnumHandler}  
  3. </select>  

查询2：

[java] view plain copy
  1. User selectById(@Param("id") int id);  

[html] view plain copy
  1. <select id="selectById" resultMap="BaseResultMap">  
  2.     select * from user_enum where id = #{id}  
  3. </select>  
  4. <resultMap id="BaseResultMap" type="User">  
  5.     <id column="id" property="id"/>  
  6.     <result column="name" property="name"/>  
  7.     <result column="password" property="password"/>  
  8.     <result column="user_type" property="userType" typeHandler="com.chrhc.mybatis.handler.EnumHandler"/>  
  9. </resultMap>  

插入:

[java] view plain copy
  1. Integer insertUser(User user);  

[html] view plain copy
  1. <insert id="insertUser" parameterType="User">  
  2.     insert into user_enum(id, name, password, user_type)values(#{id}, #{name}, #{password}, #{userType,typeHandler=com.chrhc.mybatis.handler.EnumHandler})  
  3. </insert>  

用到的枚举：

[java] view plain copy
  1. public enum UserType implements Identifiable<Integer> {  
  2.       
  3.     TEACHER(1), STUDENT(2);  
  4.       
  5.     private int type;  
  6.       
  7.     private UserType(int type){  
  8.         this.type = type;  
  9.     }  
  10.       
  11.     @Override  
  12.     public Integer getId(){  
  13.         return this.type;  
  14.     }  
  15.       
  16. }  

所需要做的仅仅是让你的枚举实现Identifiable接口就可以了。


github地址：https://github.com/xjs1919/enumhandler

------------------------------------------

如果不做其他处理，想在mapper.xml中使用我们自定义的枚举处理器那么就需要在每一个用到枚举的地方都得加一个typeHandler="com.chrhc.mybatis.handler.EnumHandler",相当的不方便。一个更方便的做法是提前把这些枚举都注册好，xml中就干净多了。

配置我们自定义的SqlSessionFactoryBean：

[html] view plain copy
  1. <!-- mybatis相关 -->  
  2.     <!-- sqlSessionFactory -->  
  3.     <bean id="sqlSessionFactory" class="com.weike.pc.spring.ScanEnumSqlSessionFactoryBean">  
  4.         <property name="enumBasePackage" value="com.weike.pc.dao.domain"/>  
  5.         <property name="aliasBasePackage" value="com.weike.pc.dao.domain"/>  
  6.         <property name="dataSource" ref="dataSource" />  
  7.         <property name="configLocation" value="classpath:mybatis-config.xml" />  
  8.         <property name="transactionFactory" ref="transactionFactory"/>  
  9.         <property name="mapperLocations">  
  10.             <list>  
  11.                 <value>classpath*:com/weike/pc/dao/*Mapper.xml</value>  
  12.             </list>  
  13.         </property>  
  14.         <property name="plugins">  
  15.             <array>  
  16.                 <bean class="com.github.pagehelper.PageHelper">  
  17.                     <property name="properties">  
  18.                       <value>  
  19.                         dialect=mysql  
  20.                       </value>  
  21.                     </property>  
  22.                 </bean>  
  23.             </array>  
  24.         </property>  
  25.     </bean>  
[java] view plain copy
  1. // 只需要把SqlSessionFactoryBean的源码复制一份，重命名成ScanEnumSqlSessionFactoryBean，然后稍加修改：  
  2. public class ScanEnumSqlSessionFactoryBean extends SqlSessionFactoryBean implements FactoryBean<SqlSessionFactory>, InitializingBean, ApplicationListener<ApplicationEvent> {  
  3.       //我们自定义的属性，扫描Enum的base package  
  4.       private String enumBasePackage;  
  5.       //我们自定义的属性，扫描Alias的base package  
  6.       private String aliasBasePackage;  
  7.         
  8.       public void setEnumBasePackage(String enumBasePackage) {  
  9.           this.enumBasePackage = enumBasePackage;  
  10.       }  
  11.   
  12.       public void setAliasBasePackage(String aliaseBasePackage) {  
  13.             this.aliasBasePackage = aliaseBasePackage;  
  14.       }  
  15.       protected SqlSessionFactory buildSqlSessionFactory() throws IOException {  
  16.         .....................  
  17.         //alias  
  18.         if (hasLength(this.aliasBasePackage)) {  
  19.             ResolverUtil<Class<?>> resolverUtil = new ResolverUtil<Class<?>>();  
  20.             resolverUtil.find(new ResolverUtil.IsA(BaseEntity.class), this.aliasBasePackage);  
  21.             Set<Class<? extends Class<?>>> mTypes = resolverUtil.getClasses();  
  22.             for (Class<?> javaTypeClass : mTypes) {  
  23.                  logger.debug("Registered Alias : '" + javaTypeClass.getName() + "'");  
  24.                  configuration.getTypeAliasRegistry().registerAlias(javaTypeClass);  
  25.             }  
  26.         }  
  27.           
  28.         //enum  
  29.         if (hasLength(this.enumBasePackage)) {  
  30.             ResolverUtil<Class<?>> resolverUtil = new ResolverUtil<Class<?>>();  
  31.             resolverUtil.find(new ResolverUtil.IsA(Identifiable.class), this.enumBasePackage);  
  32.             Set<Class<? extends Class<?>>> mTypes = resolverUtil.getClasses();  
  33.             for (Class<?> javaTypeClass : mTypes) {  
  34.                  logger.debug("Registered Enum type handler: '" + javaTypeClass.getName() + "'->"+CommonEnumHandler.class.getName());  
  35.                 configuration.getTypeHandlerRegistry().register(javaTypeClass, CommonEnumHandler.class);  
  36.             }  
  37.         }  
  38.         .....................  
  39. }
