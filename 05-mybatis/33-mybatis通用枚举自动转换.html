定义枚举接口

为什么要定义枚举的接口呢，定义接口的好处，其实主要目的是为了在万能枚举转换器中方便泛型的使用，而且还可以规范枚举类的实现。 copy

定义万能枚举转换处理器

  1. package net.itaem.handler;  
  2.   
  3. import java.sql.CallableStatement;  
  4. import java.sql.PreparedStatement;  
  5. import java.sql.ResultSet;  
  6. import java.sql.SQLException;  
  7.   
  8. import net.itaem.less.BaseEnum;  
  9.   
  10. import org.apache.ibatis.type.BaseTypeHandler;  
  11. import org.apache.ibatis.type.JdbcType;  
  12.   
  13. /** 
  14.  * @author： Fighter168 
  15.  */  
  16. public final class UniversalEnumHandler<E extends BaseEnum> extends BaseTypeHandler<E> {  
  17.   
  18.     private Class<E> type;  
  19.     private E [] enums;  
  20.       
  21.     /** 
  22.      * 设置配置文件设置的转换类以及枚举类内容，供其他方法更便捷高效的实现 
  23.      * @param type 配置文件中设置的转换类 
  24.      */  
  25.     public UniversalEnumHandler(Class<E> type) {  
  26.         if (type == null)  
  27.             throw new IllegalArgumentException("Type argument cannot be null");  
  28.         this.type = type;  
  29.         this.enums = type.getEnumConstants();  
  30.         if (this.enums == null)  
  31.             throw new IllegalArgumentException(type.getSimpleName()  
  32.                     + " does not represent an enum type.");  
  33.     }  
  34.       
  35.     @Override  
  36.     public void setNonNullParameter(PreparedStatement ps, int i, E parameter,  
  37.             JdbcType jdbcType) throws SQLException {  
  38.         //BaseTypeHandler已经帮我们做了parameter的null判断  
  39.         ps.setObject(i,(String)parameter.getValue(), jdbcType.TYPE_CODE);   
  40.     }  
  41.   
  42.     @Override  
  43.     public E getNullableResult(ResultSet rs, String columnName)  
  44.             throws SQLException {  
  45.         // 根据数据库存储类型决定获取类型，本例子中数据库中存放String类型  
  46.         String i = rs.getString(columnName);  
  47.         if (rs.wasNull()) {  
  48.             return null;  
  49.         } else {  
  50.             // 根据数据库中的value值，定位PersonType子类  
  51.             return locateEnumStatus(i);  
  52.         }  
  53.     }  
  54.   
  55.     @Override  
  56.     public E getNullableResult(ResultSet rs, int columnIndex)  
  57.             throws SQLException {  
  58.         // 根据数据库存储类型决定获取类型，本例子中数据库中存放String类型  
  59.         String i = rs.getString(columnIndex);  
  60.         if (rs.wasNull()) {  
  61.             return null;  
  62.         } else {  
  63.             // 根据数据库中的value值，定位PersonType子类  
  64.             return locateEnumStatus(i);  
  65.         }  
  66.     }  
  67.   
  68.     @Override  
  69.     public E getNullableResult(CallableStatement cs, int columnIndex)  
  70.             throws SQLException {  
  71.         // 根据数据库存储类型决定获取类型，本例子中数据库中存放String类型  
  72.         String i = cs.getString(columnIndex);  
  73.         if (cs.wasNull()) {  
  74.             return null;  
  75.         } else {  
  76.             // 根据数据库中的value值，定位PersonType子类  
  77.             return locateEnumStatus(i);  
  78.         }  
  79.     }  
  80.   
  81.      /** 
  82.      * 枚举类型转换，由于构造函数获取了枚举的子类enums，让遍历更加高效快捷 
  83.      * @param value 数据库中存储的自定义value属性 
  84.      * @return value对应的枚举类 
  85.      */  
  86.     private E locateEnumStatus(String value) {  
  87.         for(E e : enums) {  
  88.             if(e.getValue().equals(value)) {  
  89.                 return e;  
  90.             }  
  91.         }  
  92.         throw new IllegalArgumentException("未知的枚举类型：" + value + ",请核对" + type.getSimpleName());  
  93.     }  
  94. }  

测试

           接下来，我们就修改原来的例子（上一篇博客的例子：mybatis枚举自动转换实现），只需修改typeHandler，修改成
  1. <typeHandler handler="net.itaem.handler.UniversalHandler" javaType="net.itaem.less.PersonType" jdbcType="CHAR"/>  
然后我们再次运行测试用例，会发现，我们的查询得到的结果还是一样，这样，我们就不需要为每一个枚举创建一个Handler去自动转换数据库中的枚举了。
