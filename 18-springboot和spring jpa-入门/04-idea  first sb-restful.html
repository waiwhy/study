package top.qjyoung.firstboot.web;

import org.springframework.web.bind.annotation.*;
import top.qjyoung.firstboot.pojo.Book;

import java.util.*;

/**
 * Created by QJY on 2018/1/28.
 */
@RestController
public class BookContoller {
    static Map books = new HashMap<Integer, Book>();
    
    static {
        //10.for
        for (int i = 0; i < 10; i++) {
            books.put(i, new Book(i, "book" + i, "publisher" + i));
        }
    }
    
    /**
     * 获得列表
     *
     * @return
     */
    @GetMapping("books")
    public List<Book> getBooks() {
        return new ArrayList(books.values());
    }
    
    /**
     * 新增一本书
     *
     * @param book
     * @return
     */
    @PostMapping("books")
    public String addBook(Book book) {
        books.put(book.getId(), book);
        return "success";
    }
    
    /**
     * 更新一本书
     *
     * @param book
     * @param id
     * @return
     */
    @PutMapping("books/{id}")
    public String updateBook(Book book, @PathVariable("id") Integer id) {
        books.put(id, book);
        return "success";
    }

    /**
     * 删除一本书
     *
     * @param id
     * @return
     */
    @DeleteMapping("books/{id}")
    public String deleteBook(@PathVariable("id") Integer id) {
        books.remove(id);
        return "success";
    }
}
