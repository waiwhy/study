1、创建数据表
注意：数据库名不要加 中划线“-”，否设置<property name="hibernate.hbm2ddl.auto">update</property>
时无法执行修改表字段的sql语句



Create Table

CREATE TABLE `t_customer` (
  `c_id` bigint(20) NOT NULL AUTO_INCREMENT,
  `c_name` varchar(20) NOT NULL,
  `c_gender` char(1) DEFAULT NULL,
  `c_age` int(11) DEFAULT NULL,
  `c_level` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`c_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8

5、配置映射文件
规则： 
（1）命名：实体类名.hbm.xml
（2）位置：和实体类放在同一个目录下
头参考：
 
 
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE hibernate-mapping PUBLIC 
    "-//Hibernate/Hibernate Mapping DTD 3.0//EN"
    "http://www.hibernate.org/dtd/hibernate-mapping-3.0.dtd">
<hibernate-mapping>
    <!-- 类和表的映射 -->
    <class name="com.qfedu.hibernate.pojo.Customer" table="t_customer">
        <!-- 主键 -->
        <id name="id" column="c_id">
            <!-- 主键策略 -->
            <generator class="native"></generator>
        </id>
        <!-- 其他属性映射 -->
        <property name="name" column="c_name"></property>
        <property name="gender" column="c_gender"></property>
        <property name="age" column="c_age"></property>
        <property name="level" column="c_level"></property>
    </class>
</hibernate-mapping>    

6、配置核心配置文件
规则：
（1）命名：hibernate.cfg.xml
（2）位置：源码目录的根目录（maven项目放在resources下）
头参考：

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE hibernate-configuration PUBLIC
    "-//Hibernate/Hibernate Configuration DTD 3.0//EN"
    "http://www.hibernate.org/dtd/hibernate-configuration-3.0.dtd">
<hibernate-configuration>
    
    <session-factory>
        <!-- 数据库连接 -->
        <property name="hibernate.connection.driver_class">com.mysql.jdbc.Driver</property>
        <property name="hibernate.connection.username">root</property>
        <property name="hibernate.connection.password">123456</property>
        <property name="hibernate.connection.url">jdbc:mysql://localhost:3306/hibernate-gp1701</property>
    
        <!-- 数据库方言 -->
        <property name="hibernate.dialect">org.hibernate.dialect.MySQL5InnoDBDialect</property>
        
        <!-- 可选配置 -->
        <property name="hibernate.show_sql">true</property>
        <property name="hibernate.format_sql">true</property>
        <property name="hibernate.hbm2ddl.auto">update</property>
        
        <!-- mapping文件 -->
        <mapping resource="com/qfedu/hibernate/pojo/Customer.hbm.xml"/>
    
    </session-factory>
    
</hibernate-configuration>

7、测试用例
package com.qfedu.hibernate.test;
import org.hibernate.Session;
import org.hibernate.SessionFactory;
import org.hibernate.Transaction;
import org.hibernate.cfg.Configuration;
import org.junit.Test;
import com.qfedu.hibernate.pojo.Customer;
public class HibernateTest {
    @Test
    public void testSave() {
        
        Customer cust = new Customer();
        cust.setName("Helen");
        cust.setAge(18);
        cust.setGender("女");
        cust.setLevel("VIP");
        
        //读取hibernate.cfg.xml文件
        Configuration cfg = new Configuration();
        cfg.configure();
        
        //创建SessionFactory工厂
        SessionFactory sessionFactory = cfg.buildSessionFactory();
        
        //创建session
        Session session = sessionFactory.openSession();
        
        //开启事务
        Transaction tx = session.beginTransaction();
        
        //执行添加操作
        session.save(cust);
        
        //提交事务
        tx.commit();
        
        //关闭资源
        session.close();
        sessionFactory.close();
    }
}
