package com.qfedu.hibernate.test;
import org.hibernate.Session;
import org.hibernate.Transaction;
import org.junit.Test;
import com.qfedu.hibernate.pojo.Customer;
import com.qfedu.hibernate.utils.HibernateUtil;
public class CRUDTest {
    @Test
    public void testSave() {
        
        Customer cust = new Customer();
        //cust.setId("1234567");
        cust.setName("Helen");
        cust.setAge(18);
        cust.setGender('女');
        cust.setLevel("VIP");
        
        Session session = HibernateUtil.openSession();
        Transaction tx = session.beginTransaction();
        session.save(cust);
        tx.commit();
        session.close();
        
        System.out.println(cust);//save方法会自动回填生成的id
    }
    
    @Test
    public void testUpdate() {
        
        Customer cust = new Customer();
        cust.setId(1);
        cust.setName("Helen1");
        cust.setAge(181);
        cust.setGender('男');
        cust.setLevel("VIP1");
        
        Session session = HibernateUtil.openSession();
        Transaction tx = session.beginTransaction();
        session.update(cust);
        tx.commit();
        session.close();
        
    }
    
    @Test
    public void testDelete() {
        
        Customer cust = new Customer();
        cust.setId(1);
        cust.setName("");//因为影射中定义了name不能为空
        
        Session session = HibernateUtil.openSession();
        Transaction tx = session.beginTransaction();
        session.delete(cust);
        tx.commit();
        session.close();
        
    }
    
    @Test
    public void testGet() {
        Session session = HibernateUtil.openSession();
        Customer customer = session.get(Customer.class, 2);
        session.close();
        System.out.println(customer);    
    }
    
    @Test
    public void testLoad() {
        Session session = HibernateUtil.openSession();
        Customer customer = session.load(Customer.class, 2);
        System.out.println(customer);//load方法获取的对象要在session关闭前打印
        session.close();
    }
}
