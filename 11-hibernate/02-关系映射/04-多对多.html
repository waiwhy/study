             <generator class="native"></generator>
         </id>
         <property name="name" column="name"></property>
         
         <!-- 多对多映射 -->
         <!-- 
             table:中间表名
          -->
         <set name="roles" table="t_user_role" >
             <!-- 当前方在中间表的外键 -->
             <key column="user_id"/>
             <!-- column:对方在中间表的外键 -->
             <many-to-many class="Role" column="role_id"/>
         </set>
     </class>
</hibernate-mapping>
