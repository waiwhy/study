5、配置实体映射Order
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE hibernate-mapping PUBLIC 
    "-//Hibernate/Hibernate Mapping DTD 3.0//EN"
    "http://www.hibernate.org/dtd/hibernate-mapping-3.0.dtd">
    
<hibernate-mapping package="com.qfedu.hibernate.pojo.one2many">
    <!-- 类和表的映射 -->
    <class name="Order" table="t_order">
        <!-- 主键 -->
        <id name="id" column="id">
            <!-- 主键策略 -->
            <generator class="native"></generator>
        </id>
        <!-- 其他属性映射 -->
        <property name="orderno" column="orderno" length="20"></property>
        <property name="productName" column="product_name" length="100"></property>
        
        <!-- 多对一 -->
        <many-to-one name="customer" class="Customer" column="customer_id" />
    </class>
</hibernate-mapping>

8、编写测试用例
package com.qfedu.hibernate.test;
import org.hibernate.Session;
import org.hibernate.Transaction;
import org.junit.Test;
import com.qfedu.hibernate.pojo.one2many.Customer;
import com.qfedu.hibernate.pojo.one2many.Order;
import com.qfedu.hibernate.utils.HibernateUtil;
public class One2manyTest {
    /**
     * 一个客户，两个订单
     */
    @Test
    public void testCreateCustomer() {
        
        //构建数据
        Customer cust = new Customer();
        //cust.setId("1234567");
        cust.setName("Helen1234567");
        cust.setAge(18);
        cust.setGender('女');
        cust.setLevel("VIP");
        
        Order o1 = new Order();
        o1.setOrderno("20170928001");
        o1.setProductName("小米手机");
        
        Order o2 = new Order();
        o2.setOrderno("20170928002");
        o2.setProductName("小米笔记本");
        
        Session session = HibernateUtil.openSession();
        Transaction tx = session.beginTransaction();
        
        //处理数据之间的关系
        //配置双向关联
        cust.getOrders().add(o1);
        cust.getOrders().add(o2);
        o1.setCustomer(cust);
        o2.setCustomer(cust);
        
        //保存数据（保存客户和订单）
        session.save(cust);
        session.save(o1);
        session.save(o2);
        
        tx.commit();
        session.close();
    }
    
}
