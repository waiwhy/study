1、配置pojo
@Entity
@Table(name="t_customer1")
public class Customer {
    
    @Id
    @GeneratedValue(strategy=GenerationType.IDENTITY)
    @Column(name="c_id")
    private Integer id;
    
    @Column(name="c_name")
    private String name;
    
    @Column(name="c_gender")
    private Character gender;
    
    @Transient//数据库忽略
    private Integer age;
    
    @Column(name="c_level")
    private String level;

        <!-- mapping文件 -->
        <mapping class="com.qfedu.hibernate.pojo.Customer"/>
